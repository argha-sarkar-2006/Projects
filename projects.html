Project 1: File Sorter

#creating file sorter 
file_path = input("Enter the folder path: ")
user_input = input("enter your Rename file name:")
if user_input == None:
        user_input = ""
else:
     pass
import os
files=os.listdir(file_path)
files.sort()
for index, filename in enumerate(files,start=0):
        old_path=os.path.join(file_path,filename)
        _, ext = os.path.splitext(filename)
        newname = f"{user_input}_{index}{ext}"
        new_path=os.path.join(file_path,newname)
        os.rename(old_path,new_path)
        print(f"Renamed {filename} to {newname}")
